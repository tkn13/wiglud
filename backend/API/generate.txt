Endpoint: POST /api/music/generate
Purpose: To send a request for generating new music.
Request Parameters: requestId

//Request Body {
    "request_id": "0004"
    "genre": "Pop",
    "duration": 1,
    "instrument": "Guitar"
}

//Processing Response {
    "request_id": "0004"
    "genre": "Pop",
    "duration": 1,
    "instrument": "Guitar",
    "status": "Processing",
    "status_code": StatusCodes.PROCESSING,
    "message": "File generation in progress",
    "loading": true,
    "file_url": null
}

or //Error Response {
    "request_id": "0004"
    "genre": "Pop",
    "duration": 1,
    "instrument": "Guitar",
    "status": "Error",
    "status_code": StatusCodes.INTERNAL_SERVER_ERROR,
    "message": "Internal server error during file generation",
    "loading": false,
    "file_url": null
}
